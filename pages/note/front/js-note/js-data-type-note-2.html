url字符串分割成数组并返回名称 ​

js
getNameArr(urlStr) {
    let fileListObj = [];

    let urlArr = urlStr.split(',');  // 将字符串以逗号分割生成数组

    // 遍历文件集
    for(let item in urlArr) {
    urlArr[item] = urlArr[item].trim();  // 去除字符串中所有空格

    // 获取每段字符串最后一个/索引
    let index = urlArr[item].lastIndexOf("\/");

    // 遍历数据
    fileListObj.push({
        name: urlArr[item].substring(index + 1, urlArr[item].length),
        url: urlArr[item]
    });
    }

    return fileListObj
}
